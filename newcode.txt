<script>
/* THEME */
const toggle = document.getElementById("themeToggle");
const savedTheme = localStorage.getItem("theme");

if (savedTheme) {
    document.body.dataset.theme = savedTheme;
    toggle.checked = savedTheme === "light";
}

toggle.addEventListener("change", () => {
    const theme = toggle.checked ? "light" : "dark";
    document.body.dataset.theme = theme;
    localStorage.setItem("theme", theme);
});

/* MOVIE + PLAYLIST */
let currentTitle = "";
let currentYear = "";

async function searchMovie() {
    const title = document.getElementById("title").value.trim();
    const year = document.getElementById("year").value.trim();
    const error = document.getElementById("error");
    const card = document.getElementById("movieCard");
    const msg = document.getElementById("playlistMsg");

    error.textContent = "";
    msg.textContent = "";
    card.style.display = "none";

    if (!title) {
        error.textContent = "Please enter a movie title.";
        return;
    }

    try {
        const url = new URL("http://127.0.0.1:8000/movie");
        url.searchParams.set("title", title);
        if (year) url.searchParams.set("year", year);

        const res = await fetch(url);
        if (!res.ok) throw new Error();

        const data = await res.json();

        poster.src = data.Poster !== "N/A"
            ? data.Poster
            : "https://via.placeholder.com/260x390?text=No+Poster";

        movieTitle.textContent = data.Title;
        movieYear.textContent = data.Year;
        movieRated.textContent = data.Rated;
        movieRuntime.textContent = data.Runtime;
        movieLanguage.textContent = data.Language;
        moviePlot.textContent = data.Plot;
        movieRating.textContent = data.imdbRating;

        currentTitle = data.Title;
        currentYear = data.Year;

        card.style.display = "flex";
    } catch {
        error.textContent = "Movie not found or server not running.";
    }
}

async function addToPlaylist() {
    const msg = document.getElementById("playlistMsg");

    try {
        const res = await fetch(
            `http://127.0.0.1:8000/playlist/add?title=${encodeURIComponent(currentTitle)}&year=${currentYear}`,
            { method: "POST" }
        );
        const data = await res.json();

        msg.style.color = data.error ? "#ff5252" : "#00e676";
        msg.textContent = data.error || "✅ Added to playlist!";
    } catch {
        msg.style.color = "#ff5252";
        msg.textContent = "Server error.";
    }
}
